Burglar Alarm using PIR Sensor

            
float temp = 0;

void setup(){
    pinMode(A0, INPUT);
    pinMode(13, OUTPUT);
    pinMode(12, OUTPUT);
    pinMode(11, OUTPUT);
    pinMode(10, OUTPUT);
}

void loop(){
    temp = analogRead(A0);
    temp = ((temp*5/1023)-0.5)*100;
    if (temp <10){
        noTone(13);
        digitalWrite(11, HIGH);
        digitalWrite(10, LOW);
        digitalWrite(12, LOW);
    }
    else if (temp <50 && temp>=10){
        tone(13, 95);
        digitalWrite(10, HIGH);
        digitalWrite(11, LOW);
        digitalWrite(12, LOW);
    }
    else{
        tone(13, 225);
        digitalWrite(12, HIGH);
        digitalWrite(10, LOW);
        digitalWrite(11, LOW);
    }
    delay(10);
}

Burglar Alarm using IR Sensor

            
#define led 4
#define buzzer 5
#define sensor 6
int sound = 250;
void setup(){
    Serial.begin(9600);
    pinMode(sensor, INPUT);
    pinMode(led, OUTPUT);
    pinMode(buzzer, OUTPUT);
}

void loop(){
    int detect = digitalRead(sensor);
    if (detect == HIGH)
    {
        digitalWrite(led, HIGH);
        tone(buzzer, sound);
    }
    else
    {
        digitalWrite(led, LOW);
        noTone(buzzer);
    }
    delay(300);
}

Speed control of D.C. Motor

            
void setup() {
    pinMode(9, OUTPUT);
    pinMode(3, OUTPUT);
    pinMode(8, OUTPUT);
    pinMode(7, OUTPUT);
    pinMode(5, OUTPUT);
    pinMode(4, OUTPUT);	
    digitalWrite(8, LOW);
    digitalWrite(7, LOW);
    digitalWrite(5, LOW);
    digitalWrite(4, LOW);
}
void loop() {
    directionControl();
    delay(1000);
    speedControl();
    delay(1000);
}
void directionControl() {
    analogWrite(9, 255);
    analogWrite(3, 255);
    digitalWrite(8, HIGH);
    digitalWrite(7, LOW);
    digitalWrite(5, HIGH);
    digitalWrite(4, LOW);
    delay(2000);	
    digitalWrite(8, LOW);
    digitalWrite(7, HIGH);
    digitalWrite(5, LOW);
    digitalWrite(4, HIGH);
    delay(2000);	
    digitalWrite(8, LOW);
    digitalWrite(7, LOW);
    digitalWrite(5, LOW);
    digitalWrite(4, LOW);
}

void speedControl() {
    digitalWrite(8, LOW);
    digitalWrite(7, HIGH);
    digitalWrite(5, LOW);
    digitalWrite(4, HIGH);
  
    for (int i = 0; i < 256; i++) {
        analogWrite(9, i);
        analogWrite(3, i);
        delay(20);
    }
    for (int i = 255; i >= 0; --i) {
        analogWrite(9, i);
        analogWrite(3, i);
        delay(20);
    }
                
    digitalWrite(8, LOW);
    digitalWrite(7, LOW);
    digitalWrite(5, LOW);
    digitalWrite(4, LOW);
}

Temperature Display on 16×2 LCD Module 

            
#include <LiquidCrystal.h>
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
int cm;

long readUltrasonicDistance(int triggerPin, int echoPin) {
    pinMode(triggerPin, OUTPUT);
    digitalWrite(triggerPin, LOW);
    delayMicroseconds(2);
    digitalWrite(triggerPin, HIGH);
    delayMicroseconds(10);
    digitalWrite(triggerPin, LOW);
    pinMode(echoPin, INPUT);
    return pulseIn(echoPin, HIGH);
}
void setup() {
    Serial.begin(9600);
    lcd.begin(16, 2);
}

void loop() {
    cm = 0.01723 * readUltrasonicDistance(7, 7);
    inches = (cm / 2.54);
    lcd.setCursor(0, 0);
    lcd.print(inches);
    lcd.print(" in, ");
    lcd.setCursor(0, 1);
    lcd.print(cm);
    lcd.print(" cm");
    Serial.print(cm);
    delay(10);
}

Temperature Display on 16×2 LCD Module with buzzer

              
#include <LiquidCrystal.h>
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
const int sensor = A1;
float tempc;
float tempf;
float vout;
void setup(){
    pinMode(sensor, INPUT);
    pinMode(6, OUTPUT);
    Serial.begin(9600);
    lcd.begin(16, 2);
    delay(500);
}

void loop(){
    vout = analogRead(sensor);
    vout = ((vout * 5 / 1024) - 0.5) * 100;
    tempc = vout;
    tempf = (vout * 1.8) + 32;
    if (tempc < 39){
        noTone(6);
    }
    else{
        tone(6, 225);
    }
    lcd.setCursor(0, 0);
    lcd.print("in DegreeC= ");
    lcd.print(tempc);
    lcd.setCursor(0, 1);
    lcd.print("in Fahrenheit=");
    lcd.print(tempf);
    delay(1000);
}
